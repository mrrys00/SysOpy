count_lines = wc -l
składnik1 = cat /etc/passwd | wc -l
składnik2 = ls | grep '^a'
składnik3 = grep 11

składnik1             # ⇔ cat /etc/passwd | wc -l
składnik1 | składnik3 # ⇔ cat /etc/passwd | wc -l | grep 11 
składnik2             # ⇔ ls | grep '^a'
directory_content = ls -la
directory_content               # ⇔ ls -la
directory_content | count_lines # ⇔ ls -la | wc -l
